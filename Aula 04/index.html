        import math

def resolver_segundo_grau(a, b, c):
    print(f"\nEquação: {a}x² + {b}x + {c} = 0")
    
    # O coeficiente 'a' não pode ser zero em uma equação de 2º grau
    if a == 0:
        return "O coeficiente 'a' deve ser diferente de zero para ser uma equação de 2º grau."

    # 1. Cálculo do Delta
    # Delta = b² - 4ac
    delta = b**2 - 4*a*c
    print(f"Delta (Δ) = {delta}")

    # 2. Verificação dos casos de Delta
    if delta > 0:
        x1 = (-b + math.sqrt(delta)) / (2 * a)
        x2 = (-b - math.sqrt(delta)) / (2 * a)
        return f"A equação possui duas raízes reais: x1 = {x1:.2f} e x2 = {x2:.2f}"
    
    elif delta == 0:
        x = -b / (2 * a)
        return f"A equação possui uma raiz real: x = {x:.2f}"
    
    else:
        # Para raízes complexas
        parte_real = -b / (2 * a)
        parte_imag = math.sqrt(abs(delta)) / (2 * a)
        return (f"A equação possui raízes complexas: "
                f"x1 = {parte_real:.2f} + {parte_imag:.2f}i e "
                f"x2 = {parte_real:.2f} - {parte_imag:.2f}i")

# --- Teste do programa ---
try:
    a_val = float(input("Digite o valor de a: "))
    b_val = float(input("Digite o valor de b: "))
    c_val = float(input("Digite o valor de c: "))

    resultado = resolver_segundo_grau(a_val, b_val, c_val)
    print(resultado)
except ValueError:
    print("Por favor, digite apenas números válidos.")
